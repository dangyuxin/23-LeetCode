class Solution {
    public int hammingWeight(int n) {
        int nums = 0;
        int i = 0;
        while ((n >> i) > 0) {
            if (((n >> i) & 1) == 1) {
                nums++;
            }
            i++;
        }
        return nums;
    }
}
